WGSL Compute Shader (Simplified)
@group(0) @binding(0) var<storage, read_write> particles: array<Particle>;
@group(0) @binding(1) var<uniform> params: Params;

struct Particle {
    pos: vec2<f32>,
    vel: vec2<f32>,
}

@compute @workgroup_size(64)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
    let i = gid.x;
    if (i >= arrayLength(&particles)) { return; }
    
    var p = particles[i];
    
    // Attraction to mouse + fixed attractors
    let mouseForce = params.mousePos - p.pos;
    let dist = max(length(mouseForce), 12.0);
    p.vel += normalize(mouseForce) * (params.attraction / (dist * 0.8));
    
    // Mutual repulsion (simplified)
    // ... (full version has spatial hashing)
    
    p.vel *= params.damping;
    p.pos += p.vel * 0.016;
    
    // Boundary wrap
    if (p.pos.x < -1.0) { p.pos.x = 1.0; }
    if (p.pos.x >  1.0) { p.pos.x = -1.0; }
    
    particles[i] = p;
}
Full source includes ping-pong buffers, 8 attractors, Perlin noise turbulence, and instanced quad rendering.
